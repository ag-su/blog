num_classes = 20

model = torch.hub.load('pytorch/vision:v0.9.0', 'mobilenet_v2', pretrained=True)

# ë§ˆì§€ë§‰ FC ë ˆì´ì–´ë¥¼ ìš°ë¦¬ ë°ì´í„°ì…‹ì— ë§ê²Œ ì¡°ì •í•´ì£¼ê¸° 
num_ftrs = model.classifier[-1].in_features

# loss function ê³¼ optimizer ë¥¼ ì„¤ì • 
criterion = nn.CrossEntropyLoss()
optimizer = optim.Adam(model.parameters(), lr=learning_rate)


# ì…ë ¥ ì´ë¯¸ì§€ ë°ì´í„°ì…‹ ì „ì²˜ë¦¬ (ì‚¬ì´ì¦ˆ 224, 224, tensor íƒ€ì… ë³€ê²½, ìŠ¤ì¼€ì¼ í‘œì¤€í™”)
data_transforms = transforms.Compose([
    transforms.Resize((224, 224)),
    transforms.ToTensor(),
    transforms.Normalize(mean=[0.485, 0.456, 0.406], std=[0.229, 0.224, 0.225])
])



# ë°ì´í„°ì…‹ ê²½ë¡œ ì„¤ì • ë° ë°ì´í„°ì…‹ ë¶ˆëŸ¬ì˜¤ê¸° 
path_train_img = "data/training/image_class/"
path_valid_img = "data/validation/image_class/"

train_dataset = ImageFolder(root=path_train_img, transform=data_transforms) # ë°ì´í„° ì¦ê°• O
test_dataset = ImageFolder(root=path_valid_img, transform=data_transforms) # test_dataset -> val_dataset, ë°ì´í„° ì¦ê°• X
train_loader = DataLoader(dataset=train_dataset, batch_size=batch_size, shuffle=True)

# TensorBoard writerë¥¼ í†µí•´ í•™ìŠµê³¼ì •ì„ ì‹œê°í™”í•´ë³´ê¸°
writer = SummaryWriter()

# ëª¨ë¸ í•™ìŠµ
total_step = len(train_loader)
start_time = time.time()
for epoch in range(num_epochs):
    model.train() # í•™ìŠµ ëª¨ë“œë¡œ ì„¤ì • 
    for i, (images, labels) in tqdm(enumerate(train_loader), total=len(train_loader), leave=False):
        images = images.to(device)
        labels = labels.to(device)
        
        # ìˆœì „íŒŒ
        outputs = model(images)
        loss = criterion(outputs, labels)

        # ì—­ì „íŒŒ ë° optimize
        optimizer.zero_grad()
        loss.backward()
        optimizer.step()

        # TensorBoardì—ì„œ accuracy ì™€ loss ì¶”ì   
        _, predicted = torch.max(outputs.data, 1)
        total = labels.size(0)
        correct = (predicted == labels).sum().item()
        acc = 100 * correct / total
        writer.add_scalar('Train/Loss', loss.item(), epoch * total_step + i)
        writer.add_scalar('Train/Accuracy', acc, epoch * total_step + i)
    writer.flush()
    
    
    model.eval() # í‰ê°€ ëª¨ë“œë¡œ ì„¤ì • 
    # validation ë°ì´í„°ì…‹ìœ¼ë¡œ ëª¨ë¸ í‰ê°€í•˜ê¸° 
    with torch.no_grad():
        correct = 0
        total = 0
        for i, (images, labels) in tqdm(enumerate(test_loader), total=len(test_loader), leave=False):
            images = images.to(device)
            labels = labels.to(device)
            outputs = model(images)
            _, predicted = torch.max(outputs.data, 1)
            total += labels.size(0)
            correct += (predicted == labels).sum().item()

        test_acc = 100 * correct / total
        writer.add_scalar('Test/Accuracy', test_acc, epoch)
        writer.flush()

    print(f'Epoch [{epoch+1}/{num_epochs}], Loss: {loss.item():.4f}, Training Accuracy: {acc:.2f}%, Test Accuracy: {test_acc:.2f}%')

end_time = time.time()
print(f'Training complete in {(end_time - start_time):.0f} seconds')


# ëª¨ë¸ ì •ë³´ê°€ ë‹´ê¸´ ê°€ì¤‘ì¹˜ íŒŒì¼ì„ ckpt í™•ì¥ìë¡œ ì €ì¥
